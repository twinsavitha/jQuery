AJAX  


$.ajax(url,{

    type        : "Default: GET|POST|PUT|DELETE",

    dataType    : "html|json|xml",

    success     : function(data){
        console.log("Successful Request");
    },

    error       : function(data){
        console.log("Oops!! Something went wrong");
    },

    beforeSend: function(){
        console.log("Executed before the ajax request");
    },

    complete    : function(){
        console.log("Executed when the call is completed");
    }
});
